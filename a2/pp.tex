% ==================== CUSTOMER PROBLEM ====================
\section{Customer Problem and Needs}
\begin{frame}{Customer Problem and Needs}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{block}{The Problem}
                Current solutions utilize an upload/download architecture, requiring a full upload to the cloud/server before a recipient can initiate the download, creating significant latency in large file transfers.
            \end{block}
        \end{column}

        \begin{column}{0.48\textwidth}
            \small
            \vspace{0.2cm}
            \textbf{Key Core Pain Points:}
            \vspace{0.2cm}
            \begin{itemize}
                \item \textbf{Security Vulnerabilities:} Centralized servers are prone to data breaches and unauthorized access.
                \item \textbf{File Size Limitations:} Existing free tiers heavily restrict large file uploads.
                \item \textbf{Privacy Concerns:} Data logging and tracking by major cloud providers.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

% ==================== SOLUTION ====================
\section{Solution}
\begin{frame}{The SendIt Solution}
    \begin{block}{The Value Proposition}
        SendIt establishes a real-time, encrypted tunnel that allows recipients to begin downloading massive files the moment the sender hits start, bypassing the cloud entirely.
    \end{block}

    \vspace{0.3cm}
    \begin{columns}[t]
        \begin{column}{0.32\textwidth}
            \small \textbf{P2P Architecture}\\
            Eliminates the ``double-wait'' by streaming data directly between devices via WebRTC.
        \end{column}
        \begin{column}{0.32\textwidth}
            \small \textbf{Zero-Knowledge}\\
            Files never touch a central server, removing the risk of third-party breaches.
        \end{column}
        \begin{column}{0.32\textwidth}
            \small \textbf{Pure Scalability}\\
            Bypass storage caps and bandwidth throttling by utilizing user-owned hardware.
        \end{column}
    \end{columns}
\end{frame}
