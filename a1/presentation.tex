\begin{frame}{The Problem}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{SME side}
      \begin{itemize}
        \item Small tasks (10--20~h) with no in-house resource
        \item Cannot afford a full-time hire
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Student side}
      \begin{itemize}
        \item Hard to find real-world experience
        \item Traditional internships are scarce
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{1em}
  \centering
  \textit{Two pain points -- one marketplace.}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{The Solution -- StudentBridge}
  \begin{itemize}
    \item Online platform for \textbf{short, paid mini-projects} ($\sim$\,10~h)
    \item SMEs post scoped tasks; students apply and deliver
    \item Student receives \textbf{300--500~SEK} + verified \emph{Certificate of Work}
    \item Platform earns a \textbf{15\% fee} per transaction
  \end{itemize}
  \vspace{0.5em}
  \begin{block}{Unique Value Proposition}
    CV-worthy experience for students; affordable, fast delivery for SMEs.
  \end{block}
\end{frame}
